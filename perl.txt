$ref = {anonymous hash};

use
      @array: @{$ref}
       %hash: %{$ref}
   $array[3]: ${$ref}[3]     or $ref->[3]*
$hash{'key'}: ${$ref}{'key'} or $ref->{'key'}

*between 2 subscripts, -> is optional:
@array = ([...], [...], [...]);
${$array[1]}[2] <=> $array[1]->[2] <=> $array[1][2] (3rd elem from 2nd array)
